    static dispatch_once_t onceToken;
    dispatch_once(&onceToken, ^{
    NSString selString = @”pushViewController:animated:”;

    Method originalMethod = class_getInstanceMethod(self, NSSelectorFromString(selString));

  • (void)xb_pushViewController:(UIViewController *)viewController animated:(BOOL)animated {

测试结果发现：在push的同时控制台打印“push好了”，这是不行的，因为上述push是有动画效果的，那么从开始push到整个页面完全展示是需要时间的；然而在push的时候就已经执行了打印方法。因此说明在
